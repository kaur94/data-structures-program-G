using namespace std;
 
struct Node
{
    int data;
    Node *next;
}*front=NULL, *rear=NULL,*p,*q;
 
Node *newnode(int x)
{
    q=new Node;
    q->data=x;
    q->next=NULL;
    return(q);
}
 
void enqueue(Node *eq)
{
    if(front==NULL)

	{
		
        front=rear=eq;
        front->next=NULL;
    }
    
    
    else
    {
        rear->next=eq;
        rear=eq;
        eq->next=NULL;
        
    }
}
 
void dequeue(){
    if(front==NULL){

    else{
    	p=front;
    	front=front->next;

        delete(p);

    }
}
